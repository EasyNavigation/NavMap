207class LayerRegistry {
208public:
221 template<typename T>
222 std::shared_ptr<LayerView<T>> add_or_get(
223 const std::string & name,
224 size_t nitems,
225 LayerType type)
226 {
227 auto it = layers_.find(name);
228 if (it != layers_.end()) {
229 return std::dynamic_pointer_cast<LayerView<T>>(it->second);
230 }
231 auto view = std::make_shared<LayerView<T>>(name, nitems, type);
232 layers_[name] = view;
233 return view;
234 }
235
241 std::shared_ptr<LayerViewBase> get(const std::string & name) const
242 {
243 auto it = layers_.find(name);
244 return it == layers_.end() ? nullptr : it->second;
245 }
246
251 std::vector<std::string> list() const
252 {
253 std::vector<std::string> out;
254 out.reserve(layers_.size());
255 for (const auto & kv : layers_) {
256 out.push_back(kv.first);
257 }
258 return out;
259 }
260
265 bool remove(const std::string & name)
266 {
267 return layers_.erase(name) > 0;
268 }
269
277 void resize_all(size_t nitems)
278 {
279 for (auto & kv : layers_) {
280 if (auto v = std::dynamic_pointer_cast<LayerView<uint8_t>>(kv.second)) {
281 v->data_.resize(nitems);
282 continue;
283 }
284 if (auto v = std::dynamic_pointer_cast<LayerView<float>>(kv.second)) {
285 v->data_.resize(nitems);
286 continue;
287 }
288 if (auto v = std::dynamic_pointer_cast<LayerView<double>>(kv.second)) {
289 v->data_.resize(nitems);
290 continue;
291 }
292 }
293 }
294
295private:
296 std::unordered_map<std::string, std::shared_ptr<LayerViewBase>> layers_;
297};

795 template<typename T>
796 std::shared_ptr<LayerView<T>> add_layer(
797 const std::string & name,
798 const std::string & description,
799 const std::string & unit,
800 const std::string & src_layer)
801 {
802 auto dst = layers.add_or_get<T>(name, navcels.size(), layer_type_tag<T>());
803 if (dst->data().size() != navcels.size()) {
804 dst->data().assign(navcels.size(), T{});
805 }
806 layer_meta[name] = LayerMeta{description, unit, true};
807
808 if (!has_layer(src_layer)) {
809 return dst;
810 }
811
812 if (layer_copy<T>(src_layer, name)) {
813 return dst;
814 }
815
816 const std::size_t n = dst->size();
817 for (NavCelId cid = 0; cid < n; ++cid) {
818 double v = layer_get_as_double(src_layer, cid);
819 if (std::isnan(v)) {v = 0.0;}
820
821 if constexpr (std::is_floating_point_v<T>) {
822 dst->data()[cid] = static_cast<T>(v);
823 } else if constexpr (std::is_integral_v<T>) {
824 double lo = 0.0;
825 double hi = static_cast<double>(std::numeric_limits<T>::max());
826 if (v < lo) {v = lo;}
827 if (v > hi) {v = hi;}
828 dst->data()[cid] = static_cast<T>(std::llround(v));
829 } else {
830 static_assert(sizeof(T) == 0, "Unsupported layer type");
831 }
832 }
833
834 return dst;
835 }
