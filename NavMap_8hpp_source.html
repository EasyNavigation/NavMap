147{

184class LayerRegistry {
185public:
198 template<typename T>
199 std::shared_ptr<LayerView<T>> add_or_get(
200 const std::string & name,
201 size_t nitems,
202 LayerType type)
203 {
204 auto it = layers_.find(name);
205 if (it != layers_.end()) {
206 return std::dynamic_pointer_cast<LayerView<T>>(it->second);
207 }
208 auto view = std::make_shared<LayerView<T>>(name, nitems, type);
209 layers_[name] = view;
210 return view;
211 }
212
218 std::shared_ptr<LayerViewBase> get(const std::string & name) const
219 {
220 auto it = layers_.find(name);
221 return it == layers_.end() ? nullptr : it->second;
222 }
223
228 std::vector<std::string> list() const
229 {
230 std::vector<std::string> out;
231 out.reserve(layers_.size());
232 for (const auto & kv : layers_) {
233 out.push_back(kv.first);
234 }
235 return out;
236 }
237
245 void resize_all(size_t nitems)
246 {
247 for (auto & kv : layers_) {
248 if (auto v = std::dynamic_pointer_cast<LayerView<uint8_t>>(kv.second)) {
249 v->data_.resize(nitems);
250 continue;
251 }
252 if (auto v = std::dynamic_pointer_cast<LayerView<float>>(kv.second)) {
253 v->data_.resize(nitems);
254 continue;
255 }
256 if (auto v = std::dynamic_pointer_cast<LayerView<double>>(kv.second)) {
257 v->data_.resize(nitems);
258 continue;
259 }
260 }
261 }
262
263private:
264 std::unordered_map<std::string, std::shared_ptr<LayerViewBase>> layers_;
265};
