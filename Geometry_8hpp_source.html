81inline Vec3 triangle_normal(const Vec3 & a, const Vec3 & b, const Vec3 & c)
82{
83 Vec3 n = (b - a).cross(c - a);
84 float len = n.norm();
85 return len > 1e-12f ? n / len : Vec3(0.0f, 0.0f, 1.0f);
86}

109inline bool ray_triangle_intersect(
110 const Vec3 & orig,
111 const Vec3 & dir,
112 const Vec3 & v0,
113 const Vec3 & v1,
114 const Vec3 & v2,
115 float & t,
116 float & u,
117 float & v)
118{
119 const float kEps = 1e-8f;
120 Vec3 e1 = v1 - v0;
121 Vec3 e2 = v2 - v0;
122 Vec3 pvec = dir.cross(e2);
123 float det = e1.dot(pvec);
124 if (std::fabs(det) < kEps) {
125 return false;
126 }
127 float inv_det = 1.0f / det;
128 Vec3 tvec = orig - v0;
129 u = tvec.dot(pvec) * inv_det;
130 if (u < -kEps || u > 1.0f + kEps) {
131 return false;
132 }
133 Vec3 qvec = tvec.cross(e1);
134 v = dir.dot(qvec) * inv_det;
135 if (v < -kEps || (u + v) > 1.0f + kEps) {
136 return false;
137 }
138 t = e2.dot(qvec) * inv_det;
139 return t > kEps;
140}

194 inline int longest_axis() const
195 {
196 Vec3 d = max - min;
197 if (d.x() >= d.y() && d.x() >= d.z()) {return 0;}
198 if (d.y() >= d.x() && d.y() >= d.z()) {return 1;}
199 return 2;

213 inline bool intersects_ray(
214 const Vec3 & o,
215 const Vec3 & d,
216 float tmax = 1e30f) const
217 {
218 const float kEps = 1e-12f;
219 float tmin_local = 0.0f;
220
221 for (int i = 0; i < 3; ++i) {
222 const float di = d[i];
223 const float oi = o[i];
224 const float min_i = min[i];
225 const float max_i = max[i];
226
227 if (std::fabs(di) < kEps) {
228 // Parallel to this slab: must be inside [min_i, max_i]
229 if (oi < min_i || oi > max_i) {return false;}
230 continue;
231 }
232
233 float inv = 1.0f / di;
234 float t0 = (min_i - oi) * inv;
235 float t1 = (max_i - oi) * inv;
236 if (t0 > t1) {std::swap(t0, t1);}
237
238 if (t0 > tmin_local) {tmin_local = t0;}
239 if (t1 < tmax) {tmax = t1;}
240
241 if (tmax < tmin_local) {return false;}
242 }
243 return tmax >= tmin_local && tmax > 0.0f;
244 }

292inline Vec3 closest_point_on_triangle(
293 const Vec3 & p,
294 const Vec3 & a,
295 const Vec3 & b,
296 const Vec3 & c)
297{
298 const Vec3 ab = b - a;
299 const Vec3 ac = c - a;
300 const Vec3 ap = p - a;
301
302 float d1 = ab.dot(ap);
303 float d2 = ac.dot(ap);
304 if (d1 <= 0.0f && d2 <= 0.0f) {
305 return a;
306 }
307
308 const Vec3 bp = p - b;
309 float d3 = ab.dot(bp);
310 float d4 = ac.dot(bp);
311 if (d3 >= 0.0f && d4 <= d3) {
312 return b;
313 }
314
315 float vc = d1 * d4 - d3 * d2;
316 if (vc <= 0.0f && d1 >= 0.0f && d3 <= 0.0f) {
317 float v = d1 / (d1 - d3);
318 return a + v * ab;
319 }
320
321 const Vec3 cp = p - c;
322 float d5 = ab.dot(cp);
323 float d6 = ac.dot(cp);
324 if (d6 >= 0.0f && d5 <= d6) {
325 return c;
326 }
327
328 float vb = d5 * d2 - d1 * d6;
329 if (vb <= 0.0f && d2 >= 0.0f && d6 <= 0.0f) {
330 float w = d2 / (d2 - d6);
331 return a + w * ac;
332 }
333
334 float va = d3 * d6 - d5 * d4;
335 if (va <= 0.0f && (d4 - d3) >= 0.0f && (d5 - d6) >= 0.0f) {
336 float w = (d4 - d3) / ((d4 - d3) + (d5 - d6));
337 return b + w * (c - b);
338 }
339
340 float ab2 = ab.dot(ab);
341 float ac2 = ac.dot(ac);
342 float abac = ab.dot(ac);
343 float denom = (ab2 * ac2 - abac * abac);
344 if (std::fabs(denom) < 1e-20f) {
345 float da = (p - a).squaredNorm();
346 float db = (p - b).squaredNorm();
347 float dc = (p - c).squaredNorm();
348 if (da <= db && da <= dc) {return a;}
349 if (db <= dc) {return b;}
350 return c;
351 }
352 float v = (ac2 * d1 - abac * d2) / denom;
353 float w = (ab2 * d2 - abac * d1) / denom;
354 return a + v * ab + w * ac;
355}

370inline float point_triangle_squared_distance(
371 const Vec3 & p,
372 const Vec3 & a,
373 const Vec3 & b,
374 const Vec3 & c,
375 Vec3 * closest = nullptr)
376{
377 Vec3 q = closest_point_on_triangle(p, a, b, c);
378 if (closest) {*closest = q;}
379 return (q - p).squaredNorm();
380}
